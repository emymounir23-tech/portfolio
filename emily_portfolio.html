import re
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer

# Load Sentiment140 dataset (1.6M tweets)
df = pd.read_csv("training.1600000.processed.noemoticon.csv",
    encoding="latin-1", header=None, usecols=[0, 5])
df.columns = ["sentiment", "text"]

# Text cleaning pipeline
def clean_text(text):
    text = str(text).lower()
    text = re.sub(r"http\S+|www\S+", "", text)  # remove URLs
    text = re.sub(r"@\w+", "", text)              # remove @mentions
    text = re.sub(r"[^a-z\s]", "", text)           # letters only
    return re.sub(r"\s+", " ", text).strip()

df["clean_text"] = df["text"].apply(clean_text)

# Balance: 500K per class
neg = df[df["sentiment"]==0].sample(500000, random_state=42)
pos = df[df["sentiment"]==4].sample(500000, random_state=42)
df_bal = pd.concat([neg, pos]).sample(frac=1)

# TF-IDF vectorization (bigrams, top 50K features)
vectorizer = TfidfVectorizer(max_features=50000, ngram_range=(1,2))
X = vectorizer.fit_transform(df_bal["clean_text"])
